
CREATE TABLE FDW_SA_KOU_AUTOHOME
(
	ID                   NUMBER(10) NOT NULL ,
	THREAD_ID            NUMBER(10) NOT NULL ,
	WEB_LABEL            VARCHAR2(255) NOT NULL ,
	COLLECT_DATE         NUMBER(10) NOT NULL ,
	WAYS_VERSION_ID      NUMBER(10) NOT NULL ,
	WAYS_SUB_MODEL_ID    NUMBER(10) NOT NULL ,
	WAYS_SUB_MODEL_NAME  VARCHAR2(255) NOT NULL ,
	TARGET_MODEL         VARCHAR2(255) NOT NULL ,
	TARGET_MANF          VARCHAR2(255) NOT NULL ,
	TARGET_BRAND         VARCHAR2(255) NOT NULL ,
	TARGET_DETAIL        VARCHAR2(1000) NOT NULL ,
	WORD_VIEW            VARCHAR2(1000) NOT NULL ,
	WORD_ENV             VARCHAR2(1000) NOT NULL ,
	WORD_GRADE           VARCHAR2(1000) NOT NULL ,
	WORD_SENTI           VARCHAR2(1000) NOT NULL ,
	SCORE_TOTAL          VARCHAR2(20) NOT NULL 
);

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.ID IS '主键';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.THREAD_ID IS '帖子ID';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.WEB_LABEL IS '原文分类';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.COLLECT_DATE IS '采集日期';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.WAYS_VERSION_ID IS '型号ID';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.WAYS_SUB_MODEL_ID IS '子车型ID';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.WAYS_SUB_MODEL_NAME IS 'ways子车型';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.TARGET_MODEL IS '提及的子车型';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.TARGET_MANF IS '提及厂商';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.TARGET_BRAND IS '提及的品牌';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.TARGET_DETAIL IS '提及的子车型详细描述';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.WORD_VIEW IS '角度词';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.WORD_ENV IS '角度场景';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.WORD_GRADE IS '情感程度词';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.WORD_SENTI IS '情感主词';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME.SCORE_TOTAL IS '情感综合得分';

CREATE UNIQUE INDEX XPK用户口碑情感分析 ON FDW_SA_KOU_AUTOHOME
(ID   ASC);

ALTER TABLE FDW_SA_KOU_AUTOHOME
	ADD CONSTRAINT  XPK用户口碑情感分析 PRIMARY KEY (ID);

CREATE TABLE FDW_SA_KOU_AUTOHOME_ENV
(
	ID                   NUMBER(10) NOT NULL ,
	WORD                 VARCHAR2(1000) NOT NULL ,
	DATE                 DATE NOT NULL 
);

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME_ENV.ID IS '主键ID';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME_ENV.WORD IS '背景词';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME_ENV.DATE IS '添加日期';

CREATE UNIQUE INDEX XPK背景词表 ON FDW_SA_KOU_AUTOHOME_ENV
(ID   ASC);

ALTER TABLE FDW_SA_KOU_AUTOHOME_ENV
	ADD CONSTRAINT  XPK背景词表 PRIMARY KEY (ID);

CREATE TABLE FDW_SA_KOU_AUTOHOME_VIEW
(
	ID                   NUMBER(10) NOT NULL ,
	WORD                 VARCHAR2(1000) NOT NULL ,
	ADD_DATE                 DATE NOT NULL 
);

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME_VIEW.ID IS '主键ID';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME_VIEW.WORD IS '角度词';

COMMENT ON COLUMN FDW_SA_KOU_AUTOHOME_VIEW.ADD_DATE IS '添加日期';

CREATE UNIQUE INDEX XPK情感角度表 ON FDW_SA_KOU_AUTOHOME_VIEW
(ID   ASC);

ALTER TABLE FDW_SA_KOU_AUTOHOME_VIEW
	ADD CONSTRAINT  XPK情感角度表 PRIMARY KEY (ID);
